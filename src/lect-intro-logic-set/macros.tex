\NewDocumentCommand{\newopr}{sm}{
  \IfBooleanTF{#1}
    {\expandafter\newcommand\csname #2\endcsname{\operatorname*{#2}}}
    {\expandafter\newcommand\csname #2\endcsname{\operatorname{#2}}}
}

  \IfBooleanTF{#1}
    {\expandafter\newcommand\csname #2\endcsname{\operatorname*{\symsfup{#2}}}}
    {\expandafter\newcommand\csname #2\endcsname{\operatorname{\symsfup{#2}}}}
}
